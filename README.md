def seluruhData () :
    if (len(daftarMobil) == 0) :

#function untuk menampilkan data tertentu
def searchData(uid) :
    tandai = -1
    for x in range (len(daftarMobil)) :
        if (uid == daftarMobil[x]['PLAT']) :
            tandai = x
    return tandai

def dataTertentu () :

        print('')
    else :
        print('Mobil dengan Plat Nomor ', (uid), ' :')
        printData(lokasiindex)
